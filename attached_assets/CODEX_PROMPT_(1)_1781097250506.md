## FEATURE 1 — Booking Deletion with OTP Authorization

### Business rules

1. **Any admin** can *request* a booking deletion. They must supply a written reason and
   confirm whether it is a mistake or a client cancellation.
2. The request is saved to a new `booking_delete_requests` table with status `pending`.
3. An OTP is immediately sent to `CONFIG.ADMIN_EMAIL` using the existing
   `send-otp-email` edge function with `resource_type: 'booking_delete'` and
   `resource_id: <booking_id>`.
4. An OTP entry modal appears on the requesting admin's screen. They enter the code that
   arrives on the main admin's email.
5. On successful OTP verification the booking is **hard-deleted** from the `bookings`
   table, but only after:
   a. An audit row is inserted into `booking_edit_log`
      (`action: 'delete'`, `reason`, `admin_id`, `old_values: <full booking JSON>`).
   b. The delete request record is updated to `status: 'approved'`.
6. If the main admin **rejects** the request they update it to `status: 'rejected'` with
   a `rejection_reason`. The requesting admin should see this on their next dashboard load.
7. The main admin also sees all pending delete requests in a **Pending Deletions** card
   in the Calendar tab (or a dedicated section). They can approve (sends OTP to
   themselves) or reject inline.
8. After deletion the calendar and bookings archive are refreshed.

CREATE TABLE IF NOT EXISTS public.booking_delete_requests (
  id                  UUID        PRIMARY KEY DEFAULT gen_random_uuid(),
  booking_id          UUID        NOT NULL REFERENCES public.bookings(id) ON DELETE CASCADE,
  requested_by        UUID        NOT NULL REFERENCES public.profiles(id),
  reason              TEXT        NOT NULL,
  cancellation_type   TEXT        NOT NULL CHECK (cancellation_type IN ('mistake','client_cancelled')),
  status              TEXT        NOT NULL DEFAULT 'pending'
                                  CHECK (status IN ('pending','approved','rejected')),
  rejection_reason    TEXT,
  reviewed_by         UUID        REFERENCES public.profiles(id),
  reviewed_at         TIMESTAMPTZ,
  created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX IF NOT EXISTS idx_delete_requests_booking  ON public.booking_delete_requests(booking_id);
CREATE INDEX IF NOT EXISTS idx_delete_requests_status   ON public.booking_delete_requests(status);
CREATE INDEX IF NOT EXISTS idx_delete_requests_requester ON public.booking_delete_requests(requested_by);

ALTER TABLE public.booking_delete_requests ENABLE ROW LEVEL SECURITY;

CREATE POLICY "delete_requests_admin" ON public.booking_delete_requests
  FOR ALL USING (public.is_admin()) WITH CHECK (public.is_admin());

async function executeBookingDeletion(bookingId, deleteReqId, reason) {
  try {
    // Fetch full booking for audit log
    const { data: booking } = await sb
      .from('bookings')
      .select('*')
      .eq('id', bookingId)
      .single();

    if (!booking) { toast('Booking not found', 'error'); return; }

    // 1. Insert audit log
    await sb.from('booking_edit_log').insert({
      booking_id:  bookingId,
      admin_id:    currentProfile?.id,
      action:      'delete',
      reason,
      old_values:  booking,
      approved_at: new Date().toISOString(),
    });

    // 2. Hard delete the booking
    const { error: delError } = await sb
      .from('bookings')
      .delete()
      .eq('id', bookingId);

    if (delError) throw delError;

    // 3. Mark delete request as approved
    if (deleteReqId) {
      await sb.from('booking_delete_requests').update({
        status:      'approved',
        reviewed_by: currentProfile?.id,
        reviewed_at: new Date().toISOString(),
      }).eq('id', deleteReqId);
    }

    toast('Booking permanently deleted', 'success');
    closeModal('modal-otp-verify');
    closeModal('modal-booking');
    await renderCalendar();
    await loadBookingsArchive();
    await loadPendingDeletionsBadge();

  } catch (err) {
    toast('Deletion failed: ' + err.message, 'error');
  }
}
